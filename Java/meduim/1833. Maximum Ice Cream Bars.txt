import java.util.Arrays;
//task 1833
class Solution {
    public int maxIceCream(int[] costs, int coins) {
        // 1. Сортируем цены от дешёвых к дорогим
        Arrays.sort(costs);
        
        int count = 0;
        
        // 2. Покупаем мороженое, пока хватает монет
        for (int cost : costs) {
            if (coins >= cost) {
                coins -= cost;
                count++;
            } else {
                // Если монет меньше, чем стоит текущее мороженое,
                // дальше идти нет смысла, так как остальные еще дороже
                break;
            }
        }
        
        return count;
    }
}
