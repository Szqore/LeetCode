class Solution { //task 1871
    public boolean canReach(String s, int minJump, int maxJump) {
        int n = s.length();
        // Если последний символ равен '1', мы физически не можем туда наступить
        if (s.charAt(n - 1) == '1') {
            return false;
        }

        // dp[i] будет true, если мы можем добраться до индекса i
        boolean[] dp = new boolean[n];
        dp[0] = true; // Стартовая позиция всегда доступна

        // Переменная для хранения количества доступных точек в текущем окне прыжка
        int reachableCount = 0;

        for (int i = 1; i < n; i++) {
            // 1. Добавляем в окно элемент, который теперь стал доступен для прыжка на индекс i
            if (i >= minJump && dp[i - minJump]) {
                reachableCount++;
            }

            // 2. Удаляем из окна элемент, который вышел за пределы максимального прыжка maxJump
            if (i > maxJump && dp[i - maxJump - 1]) {
                reachableCount--;
            }

            // 3. Проверяем, можем ли мы прыгнуть в текущую точку
            if (s.charAt(i) == '0' && reachableCount > 0) {
                dp[i] = true;
            }
        }

        // Возвращаем результат для последнего индекса
        return dp[n - 1];
    }
}
