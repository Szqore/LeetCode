import java.util.HashMap;
import java.util.Map;
//task 3020
class Solution {
    public int maximumLength(int[] nums) {
        // Шаг 1: Подсчет частот всех чисел
        Map<Integer, Integer> count = new HashMap<>();
        for (int num : nums) {
            count.put(num, count.getOrDefault(num, 0) + 1);
        }

        int ans = 1;

        // Шаг 2: Обработка особого случая с числом 1
        if (count.containsKey(1)) {
            int ones = count.get(1);
            if (ones % 2 == 0) {
                ans = ones - 1; // Максимальное нечетное количество
            } else {
                ans = ones;
            }
        }

        // Шаг 3: Перебор всех возможных оснований для построения цепочек
        for (int base : count.keySet()) {
            if (base == 1) continue;

            int length = 0;
            long x = base;

            // Строим цепочку квадратов: x, x^2, x^4, ...
            while (count.containsKey((int) x) && count.get((int) x) >= 2) {
                length += 2;
                x = x * x; 
                // Предотвращаем переполнение long
                if (x > 1_000_000_000) break;
            }

            // Если элемент существует как пик (например, больше не возводим в квадрат)
            if (count.containsKey((int) x)) {
                length += 1;
            } else {
                // Если пика нет, но мы зашли в тупик, откатываем последний шаг
                length -= 1; 
            }

            ans = Math.max(ans, length);
        }

        return ans;
    }
}
