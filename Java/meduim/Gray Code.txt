import java.util.ArrayList;
import java.util.List; 
//task 89
class Solution {
    public List<Integer> grayCode(int n) {
        List<Integer> result = new ArrayList<>();
        result.add(0);

        for (int i = 0; i < n; i++) {
            int size = result.size();
            for (int j = size -1; j >= 0; j--) {
                int num = result.get(j) | (1 << i);
                result.add(num);
            }
        }
        return result;

    }
}
