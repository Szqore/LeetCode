class Solution { //task 189
    public void rotate(int[] nums, int k) {
        int n = nums.length;
        k = k % n; // На случай, если k > n
        
        //Переворачиваем весь массив
        reverse(nums, 0, n - 1);
        //Переворачиваем первую часть (0 до k-1)
        reverse(nums, 0, k - 1);
        //Переворачиваем вторую часть (k до n-1)
        reverse(nums, k, n - 1);
    }
    
    private void reverse(int[] nums, int left, int right) {
        while (left < right) {
            int temp = nums[left];
            nums[left] = nums[right];
            nums[right] = temp;
            left++;
            right--;
        }
    }
}
