class Solution { //task 1391
    boolean[][] visited;  // массив для отслеживания посещённых клеток
    boolean reached = false;  // флаг, достигли ли мы правого нижнего угла
    
    public boolean hasValidPath(int[][] grid) {
        visited = new boolean[grid.length][grid[0].length];
        dfs(grid, 0, 0);  // начинаем с левого верхнего угла
        return reached;
    }
    
    private void dfs(int[][] grid, int x, int y) {
        // если уже вышли за границы или клетка посещена
        if (x < 0 || x >= grid.length || y < 0 || y >= grid[0].length || visited[x][y]) {
            return;
        }
        
        visited[x][y] = true;  // отмечаем клетку как посещённую
        
        // если дошли до правого нижнего угла
        if (x == grid.length - 1 && y == grid[0].length - 1) {
            reached = true;
            return;
        }
        
        int cell = grid[x][y];  // текущая клетка (тип трубы)
        
        // проверяем, можем ли идти ВВЕРХ (если труба позволяет)
        if (cell == 2 || cell == 5 || cell == 6) {
            // проверяем, что клетка сверху есть и её труба соединяется с нами
            if (x > 0 && (grid[x-1][y] == 2 || grid[x-1][y] == 3 || grid[x-1][y] == 4)) {
                dfs(grid, x - 1, y);
            }
        }
        
        // проверяем, можем ли идти ВНИЗ
        if (cell == 2 || cell == 3 || cell == 4) {
            if (x + 1 < grid.length && (grid[x+1][y] == 2 || grid[x+1][y] == 5 || grid[x+1][y] == 6)) {
                dfs(grid, x + 1, y);
            }
        }
        
        // проверяем, можем ли идти ВЛЕВО
        if (cell == 1 || cell == 3 || cell == 5) {
            if (y > 0 && (grid[x][y-1] == 1 || grid[x][y-1] == 4 || grid[x][y-1] == 6)) {
                dfs(grid, x, y - 1);
            }
        }
        
        // проверяем, можем ли идти ВПРАВО
        if (cell == 1 || cell == 4 || cell == 6) {
            if (y + 1 < grid[0].length && (grid[x][y+1] == 1 || grid[x][y+1] == 3 || grid[x][y+1] == 5)) {
                dfs(grid, x, y + 1);
            }
        }
    }
}
