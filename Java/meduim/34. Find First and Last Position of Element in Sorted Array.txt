class Solution {
    public int[] searchRange(int[] nums, int target) {
        int[] result = {-1, -1};
        result[0] = findBound(nums, target, true); // Ищем левую границу
        result[1] = findBound(nums, target, false); // Ищем правую границу
        return result;
    }
//task 34
    private int findBound(int[] nums, int target, boolean isFirst) {
        int n = nums.length;
        int left = 0, right = n - 1;
        int bound = -1;

        while (left <= right) {
            int mid = left + (right - left) / 2;
            if (nums[mid] == target) {
                bound = mid;
                if (isFirst) {
                    right = mid - 1; // Ищем левее
                } else {
                    left = mid + 1; // Ищем правее
                }
            } else if (nums[mid] < target) {
                left = mid + 1;
            } else {
                right = mid - 1;
            }
        }
        return bound;
    }
}
