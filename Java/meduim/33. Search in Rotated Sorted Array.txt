class Solution {
    public int search(int[] nums, int target) {
        int left = 0, right = nums.length - 1;
        //task 33
        while (left <= right) {
            int mid = left + (right - left) / 2;
            
            if (nums[mid] == target) {
                return mid;
            }
            
            // Проверяем, отсортирована ли левая половина
            if (nums[left] <= nums[mid]) {
                // Если target находится в левой отсортированной части
                if (nums[left] <= target && target < nums[mid]) {
                    right = mid - 1;
                } else {
                    left = mid + 1;
                }
            } 
            // Иначе отсортирована правая половина
            else {
                // Если target находится в правой отсортированной части
                if (nums[mid] < target && target <= nums[right]) {
                    left = mid + 1;
                } else {
                    right = mid - 1;
                }
            }
        }
        
        return -1; // Целевое значение не найдено
    }
}
