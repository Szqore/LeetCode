class Solution { // task 1092
    public String shortestCommonSupersequence(String str1, String str2) {
        int m = str1.length();
        int n = str2.length();

        // dp[i][j] = длина НАИБОЛЬШЕЙ ОБЩЕЙ ПОДПОСЛЕДОВАТЕЛЬНОСТИ (LCS)
        // для первых i символов str1 и первых j символов str2
        // ЗАЧЕМ: LCS — это символы, которые мы добавим в ответ только ОДИН раз,
        // а не по разу из каждой строки.
        int[][] dp = new int[m + 1][n + 1];

        // Шаг 1. Находим LCS (стандартное ДП)
        for (int i = 1; i <= m; i++) {
            for (int j = 1; j <= n; j++) {
                if (str1.charAt(i - 1) == str2.charAt(j - 1)) {
                    // Символы совпадают → он войдет в LCS
                    dp[i][j] = dp[i - 1][j - 1] + 1;
                } else {
                    // Не совпадают → берем максимум из двух предыдущих вариантов
                    dp[i][j] = Math.max(dp[i - 1][j], dp[i][j - 1]);
                }
            }
        }

        // Шаг 2. Восстанавливаем САМУЮ КОРОТКУЮ ОБЩУЮ СУПЕРПОСЛЕДОВАТЕЛЬНОСТЬ (SCS)
        StringBuilder result = new StringBuilder();
        int i = m, j = n;

        // Идем с конца таблицы dp (как при восстановлении LCS, но с важным отличием)
        while (i > 0 && j > 0) {
            if (str1.charAt(i - 1) == str2.charAt(j - 1)) {
                // Символы равны → это часть LCS → добавляем ЕГО ОДИН РАЗ
                // (он покрывает и str1, и str2 одновременно)
                result.append(str1.charAt(i - 1));
                i--;
                j--;
            } 
            // Если символы НЕ равны, мы НЕ можем взять их оба сразу.
            // Мы должны добавить символ из ТОЙ строки, которая дала бОльший вклад в LCS,
            // чтобы не потерять возможность использовать общие символы позже.
            else if (dp[i - 1][j] > dp[i][j - 1]) {
                // Идем "вверх" → добавляем символ из str1 (покрываем его сейчас)
                result.append(str1.charAt(i - 1));
                i--;
            } else {
                // Идем "влево" → добавляем символ из str2
                result.append(str2.charAt(j - 1));
                j--;
            }
        }

        // Шаг 3. Добавляем оставшиеся символы из str1 (если str1 еще не закончилась)
        while (i > 0) {
            result.append(str1.charAt(i - 1));
            i--;
        }

        // Шаг 4. Добавляем оставшиеся символы из str2
        while (j > 0) {
            result.append(str2.charAt(j - 1));
            j--;
        }

        // Шаг 5. Мы собирали символы с КОНЦА (от последних к первым),
        // поэтому итоговую строку нужно перевернуть.
        return result.reverse().toString();
    }
}
