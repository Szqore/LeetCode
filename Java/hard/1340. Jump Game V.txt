class Solution { //task 1340
    public int maxJumps(int[] arr, int d) {
        int n = arr.length;
        int[] memo = new int[n];
        int maxJumps = 0;
        
        // Ищем максимальный путь, начиная с каждого возможного индекса
        for (int i = 0; i < n; i++) {
            maxJumps = Math.max(maxJumps, dfs(arr, d, i, memo));
        }
        
        return maxJumps;
    }
    
    private int dfs(int[] arr, int d, int i, int[] memo) {
        // Если значение уже вычислено, возвращаем его
        if (memo[i] != 0) {
            return memo[i];
        }
        
        int n = arr.length;
        int currentMax = 1; // Минимальный результат — остаться на текущем столбике
        
        // Прыжки вправо: от i + 1 до i + d
        for (int j = i + 1; j <= i + d && j < n; j++) {
            if (arr[j] >= arr[i]) {
                break; // Нельзя перепрыгнуть столбик, который выше или равен текущему
            }
            currentMax = Math.max(currentMax, 1 + dfs(arr, d, j, memo));
        }
        
        // Прыжки влево: от i - 1 до i - d
        for (int j = i - 1; j >= i - d && j >= 0; j--) {
            if (arr[j] >= arr[i]) {
                break; // Аналогично, путь заблокирован высоким столбиком
            }
            currentMax = Math.max(currentMax, 1 + dfs(arr, d, j, memo));
        }
        
        memo[i] = currentMax; // Сохраняем результат для мемоизации
        return currentMax;
    }
}
