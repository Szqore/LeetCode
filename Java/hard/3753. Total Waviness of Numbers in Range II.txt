import java.util.Arrays;
//task 3753
class Solution {
    // Основной метод для поиска суммарной волнистости в диапазоне [num1, num2]
    public long totalWaviness(long num1, long num2) {
        // По принципу префиксных сумм: результат для [num1, num2] равен f(num2) - f(num1 - 1)
        return countWaviness(num2) - countWaviness(num1 - 1);
    }

    // Вспомогательный метод для подсчета волнистости от 0 до limit
    private long countWaviness(long limit) {
        if (limit < 0) return 0;
        // Переводим число в массив символов для поразрядного обхода
        char[] s = String.valueOf(limit).toCharArray();
        int n = s.length;
        
        // 6-мерный массив для мемоизации:
        // 1-я ось: n - текущая позиция в числе (индекс разряда)
        // 2-я ось: 11 - предыдущая цифра `prev` (значения от -1 до 9, сдвинутые на +1 -> индексы 0-10)
        // 3-я ось: 11 - предпредыдущая цифра `prev2` (значения от -1 до 9, сдвинутые на +1 -> индексы 0-10)
        // 4-я ось: 2 - флаг `isStart` (0 или 1), показывает, идет ли еще последовательность ведущих нулей
        // 5-я ось: 2 - флаг `isLimit` (0 или 1), ограничивает ли текущая цифра выбор (чтобы не выйти за пределы `limit`)
        // 6-я ось: 2 - хранит пару: [0] -> totalWaviness, [1] -> totalCount
        long[][][][][][] memo = new long[n][11][11][2][2][2];
        
        // Заполняем массив мемоизации значениями -1 (признак того, что состояние еще не вычислено)
        for (long[][][][][] d5 : memo) {
            for (long[][][][] d4 : d5) {
                for (long[][][] d3 : d4) {
                    for (long[][] d2 : d3) {
                        for (long[] d1 : d2) {
                            Arrays.fill(d1, -1);
                        }
                    }
                }
            }
        }
        
        // Запуск рекурсивной функции Digit DP с начальными параметрами
        return dp(s, 0, -1, -1, true, true, memo)[0];
    }

    // Рекурсивная функция Digit DP
    private long[] dp(char[] s, int i, int prev, int prev2, boolean isStart, boolean isLimit, long[][][][][][] memo) {
        // Базовый случай: если дошли до конца числа, возвращаем 0 волнистости и 1 найденное число
        if (i == s.length) {
            return new long[]{0, 1}; 
        }
        
        // Сдвигаем значения цифр (-1..9) в индексы массива (0..10)
        int prevIdx = prev + 1;
        int prev2Idx = prev2 + 1;
        int startIdx = isStart ? 1 : 0;
        int limitIdx = isLimit ? 1 : 0;
        
        // Если это состояние уже было вычислено ранее, возвращаем готовый результат
        if (memo[i][prevIdx][prev2Idx][startIdx][limitIdx][0] != -1) {
            return new long[]{
                memo[i][prevIdx][prev2Idx][startIdx][limitIdx][0],
                memo[i][prevIdx][prev2Idx][startIdx][limitIdx][1]
            };
        }

        // Определяем максимально возможную цифру для текущего разряда
        // Если мы привязаны к границе исходного числа (isLimit == true), то максимум — это цифра из s[i].
        // Если мы уже свернули в меньшую сторону, то можем ставить любую цифру от 0 до 9.
        int maxDigit = isLimit ? s[i] - '0' : 9;
        
        long totalWaviness = 0;
        long totalCount = 0;

        // Перебираем все возможные цифры `d` для текущей позиции
        for (int d = 0; d <= maxDigit; d++) {
            // Определяем, останется ли число состоять из ведущих нулей на следующем шаге
            boolean nextIsStart = isStart && d == 0;
            
            // Передаем историю цифр дальше. Если это всё еще ведущий ноль, сбрасываем историю в -1.
            int nextPrev = nextIsStart ? -1 : d;
            int nextPrev2 = nextIsStart ? -1 : prev;

            int waviness = 0;
            // Проверяем, образует ли текущая тройка цифр (prev2, prev, d) "волну" (пик или долину).
            // Это возможно, только если мы вышли из режима ведущих нулей (isStart == false) 
            // и у нас уже есть две накопленные цифры в истории (prev2 != -1).
            if (!isStart && prev2 != -1) {
                if (prev2 < prev && prev > d) {      // Пик (например, 1 -> 5 -> 2)
                    waviness = 1;
                } else if (prev2 > prev && prev < d) { // Долина (например, 4 -> 1 -> 3)
                    waviness = 1;
                }
            }

            // Вычисляем флаг ограничения для следующего разряда
            boolean nextIsLimit = isLimit && d == maxDigit;
            
            // Рекурсивный вызов для следующего разряда числа
            long[] nextResult = dp(s, i + 1, nextPrev, nextPrev2, nextIsStart, nextIsLimit, memo);

            // Математика подсчета:
            // 1. Текущая точка (пик/долина) повторится ровно столько раз, сколько валидных комбинаций суффикса существует (nextResult[1]).
            // 2. К этому добавляется накопленная волнистость из всех последующих разрядов (nextResult[0]).
            totalWaviness += nextResult[0] + (long) waviness * nextResult[1];
            
            // Суммируем общее количество валидных чисел
            totalCount += nextResult[1];
        }

        // Сохраняем вычисленные значения в таблицу мемоизации
        memo[i][prevIdx][prev2Idx][startIdx][limitIdx][0] = totalWaviness;
        memo[i][prevIdx][prev2Idx][startIdx][limitIdx][1] = totalCount;

        // Возвращаем результат для текущей ветки рекурсии
        return new long[]{totalWaviness, totalCount};
    }
}
