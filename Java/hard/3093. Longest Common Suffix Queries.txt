class Solution { //task 3093
    // Класс узла Бор (Trie)
    class TrieNode {
        TrieNode[] children = new TrieNode[26];
        // Индекс слова из wordsContainer с наименьшей длиной
        int minLenWordIndex = -1; 
    }

    public int[] stringIndices(String[] wordsContainer, String[] wordsQuery) {
        TrieNode root = new TrieNode();
        
        // Переменная для хранения индекса самого короткого слова в контейнере на случай, 
        // если общего суффикса вообще нет (корневой узел)
        int globalMinLenIndex = 0;
        
        // 1. Построение дерева
        for (int i = 0; i < wordsContainer.length; i++) {
            String word = wordsContainer[i];
            
            // Обновляем глобальный минимум для корня
            if (word.length() < wordsContainer[globalMinLenIndex].length()) {
                globalMinLenIndex = i;
            }
            
            TrieNode curr = root;
            // Идем по слову с конца к началу
            for (int j = word.length() - 1; j >= 0; j--) {
                int charIdx = word.charAt(j) - 'a';
                if (curr.children[charIdx] == null) {
                    curr.children[charIdx] = new TrieNode();
                }
                
                curr = curr.children[charIdx];
                
                // Если узел новый или нашли более короткое слово (или такое же, но с меньшим индексом)
                if (curr.minLenWordIndex == -1) {
                    curr.minLenWordIndex = i;
                } else {
                    int currMinIdx = curr.minLenWordIndex;
                    if (word.length() < wordsContainer[currMinIdx].length()) {
                        curr.minLenWordIndex = i;
                    }
                }
            }
        }
        
        // Инициализируем корневой узел глобальным минимумом
        root.minLenWordIndex = globalMinLenIndex;

        // 2. Обработка запросов
        int[] result = new int[wordsQuery.length];
        for (int i = 0; i < wordsQuery.length; i++) {
            String query = wordsQuery[i];
            TrieNode curr = root;
            int lastValidIndex = root.minLenWordIndex;
            
            // Ищем суффикс, двигаясь с конца строки запроса
            for (int j = query.length() - 1; j >= 0; j--) {
                int charIdx = query.charAt(j) - 'a';
                if (curr.children[charIdx] != null) {
                    curr = curr.children[charIdx];
                    lastValidIndex = curr.minLenWordIndex;
                } else {
                    break; // Общий суффикс прервался
                }
            }
            result[i] = lastValidIndex;
        }
        
        return result;
    }
}
