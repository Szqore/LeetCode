import java.util.*;
//task 3161
class Solution {
    // По условию задачи x не превышает min(50000, 3 * queries.length)
    private static final int MAX_COORD = 50005; 
    private int[] tree;

    public List<Boolean> getResults(int[][] queries) {
        int m = MAX_COORD;
        tree = new int[4 * m];
        
        // TreeSet для хранения позиций препятствий. 0 — базовое препятствие на старте.
        TreeSet<Integer> obstacles = new TreeSet<>();
        obstacles.add(0);
        obstacles.add(m); // Ограничитель справа для корректной логики

        List<Boolean> ans = new ArrayList<>();

        for (int[] q : queries) {
            int type = q[0];
            if (type == 1) {
                int x = q[1];
                
                // Находим соседние препятствия
                int left = obstacles.floor(x);
                int right = obstacles.ceiling(x);

                // Добавляем новое препятствие
                obstacles.add(x);

                // Обновляем дерево отрезков
                update(1, 0, m, x, x - left);
                update(1, 0, m, right, right - x);
                
            } else {
                int x = q[1];
                int sz = q[2];

                // 1. Ищем максимальный промежуток среди всех целых отрезков на [0, x]
                int maxGap = query(1, 0, m, 0, x);

                // 2. Проверяем последний "хвост" перед координатой x
                int leftBoundary = obstacles.floor(x);
                int lastGap = x - leftBoundary;
                
                maxGap = Math.max(maxGap, lastGap);

                // Если нашли место под блок, возвращаем true
                ans.add(maxGap >= sz);
            }
        }
        return ans;
    }

    // Стандартная функция обновления одиночного элемента в Дереве Отрезков
    private void update(int node, int start, int end, int idx, int val) {
        if (start == end) {
            tree[node] = val;
            return;
        }
        int mid = start + (end - start) / 2;
        if (idx <= mid) {
            update(2 * node, start, mid, idx, val);
        } else {
            update(2 * node + 1, mid + 1, end, idx, val);
        }
        tree[node] = Math.max(tree[2 * node], tree[2 * node + 1]);
    }

    // Стандартный запрос на максимум на интервале [l, r]
    private int query(int node, int start, int end, int l, int r) {
        if (r < start || end < l) {
            return 0;
        }
        if (l <= start && end <= r) {
            return tree[node];
        }
        int mid = start + (end - start) / 2;
        int leftQuery = query(2 * node, start, mid, l, r);
        int rightQuery = query(2 * node + 1, mid + 1, end, l, r);
        return Math.max(leftQuery, rightQuery);
    }
}
