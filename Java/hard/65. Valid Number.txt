class Solution {
    public boolean isNumber(String s) {
        // Регулярное выражение для валидного числа
        return s.matches("^[+-]?(\\d+(\\.\\d*)?|\\.\\d+)([eE][+-]?\\d+)?$");
    }
}
// ^[+-]? — необязательный знак плюс или минус в начале.
// (\\d+(\\.\\d*)?|\\.\\d+) — левая часть числа (до экспоненты):
// \\d+(\\.\\d*)? — целое число с возможной дробной частью (например, 1, 1., 1.5).
// |\\.\\d+ — или дробная часть, начинающаяся с точки (например, .5).
// ([eE][+-]?\\d+)? — необязательная экспонента:
// [eE] — символ e или E.
// [+-]? — необязательный знак степени.
// \\d+ — целое число экспоненты.
