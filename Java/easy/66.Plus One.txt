class Solution { //task 66
    public int[] plusOne(int[] digits) {
        int n = digits.length;

        // идем с конца массива
        for (int i = n - 1; i >= 0; i--) {
            // Если цифра НЕ 9 (меньше 9)
            if (digits[i] < 9) {  // ← исправлено: digits[i] < 9
                digits[i]++;
                return digits;
            }
        //Если цифра 9, заменяем её на 0 и продолжаем (переносим 1 влево)
        digits[i] = 0;
    }
    // Если дошли сюда, значит все цифры были 9 (например, [9,9,9])
     // Создаём новый массив на 1 элемент больше
     int[] result = new int[n + 1];
     result[0] = 1;  // первый элемент 1, остальные по умолчанию 0

     return result;
    
    }
}
