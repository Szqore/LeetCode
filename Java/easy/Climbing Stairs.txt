class Solution { //task 70
    public int climbStairs(int n) {
        if (n == 1) return 1;
        
        int prev1 = 1; // dp[i-2] для i=2 это dp[0] 
        int prev2 = 2; // dp[i-1] для i=2 это dp[1]
        
        for (int i = 3; i <= n; i++) {
            int current = prev1 + prev2;
            prev1 = prev2;
            prev2 = current;
        }
        return prev2;
    }
}
