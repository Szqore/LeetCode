class Solution { //task 1979
    public int findGCD(int[] nums) {
        int min = nums[0];
        int max = nums[0];
        
        // Находим минимальный и максимальный элементы за один проход
        for (int num : nums) {
            if (num < min) {
                min = num;
            }
            if (num > max) {
                max = num;
            }
        }
        
        // Вычисляем НОД для найденных чисел
        return gcd(max, min);
    }
    
    // Алгоритм Евклида (итеративный подход)
    private int gcd(int a, int b) {
        while (b != 0) {
            int temp = b;
            b = a % b;
            a = temp;
        }
        return a;
    }
}
