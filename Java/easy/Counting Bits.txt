class Solution { //task 338
    public int[] countBits(int n) {
        int [] ans = new int[n + 1];

        for (int i = 1; i <=n; i++) {
            // i >> 1 — это i // 2
            // (i & 1) — это последний бит числа i
            ans[i] = ans [i >> 1] + (i & 1);
        }

        return ans;
    }
}
