class NumArray { //task 303
    private int[] prefix;
    
    public NumArray(int[] nums) {
        int n = nums.length;
        prefix = new int[n + 1];
        
        // prefix[i] = сумма первых i элементов (от 0 до i-1)
        for (int i = 0; i < n; i++) {
            prefix[i + 1] = prefix[i] + nums[i];
        }
    }
    
    public int sumRange(int left, int right) {
        // сумма от left до right = prefix[right+1] - prefix[left]
        return prefix[right + 1] - prefix[left];
    }
}
