/**
 * Definition for a binary tree node.
 * public class TreeNode {
 *     int val;
 *     TreeNode left;
 *     TreeNode right;
 *     TreeNode() {}
 *     TreeNode(int val) { this.val = val; }
 *     TreeNode(int val, TreeNode left, TreeNode right) {
 *         this.val = val;
 *         this.left = left;
 *         this.right = right;
 *     }
 * }
 */
class Solution { //task 110
    public boolean isBalanced(TreeNode root) {
        // Рекурсивно проверяем, сбалансировано ли дерево
        return checkHeight(root) != -1;
    }

    private int checkHeight(TreeNode node) {
        if (node == null) return 0; // Высота пустого узла — 0

        int leftHeight = checkHeight(node.left);  // Высота левого поддерева
        if (leftHeight == -1) return -1; // Левое поддерево несбалансировано

        int rightHeight = checkHeight(node.right); // Высота правого поддерева
        if (rightHeight == -1) return -1; // Правое поддерево несбалансировано

        // Если разница высот больше 1 — дерево несбалансировано
        if (Math.abs(leftHeight - rightHeight) > 1) return -1;

        // Возвращаем высоту текущего узла
        return 1 + Math.max(leftHeight, rightHeight);
    }
}
