class Solution { //task 121
    public int maxProfit(int[] prices) {
        if (prices == null || prices.length < 2) {
            return 0;
        }
        
        int minPrice = Integer.MAX_VALUE;
        int maxProfit = 0;
        
        for (int price : prices) {
            // Обновляем минимальную цену для покупки
            if (price < minPrice) {
                minPrice = price;
            }
            
            // Считаем прибыль, если продаём сегодня
            int profit = price - minPrice;
            
            // Обновляем максимальную прибыль
            if (profit > maxProfit) {
                maxProfit = profit;
            }
        }
        
        return maxProfit;
    }
}
