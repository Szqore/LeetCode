public class Solution { //task 190
    // Рассматриваем n как беззнаковое 32-битное число
    public int reverseBits(int n) {
        int result = 0;
        for (int i = 0; i < 32; i++) {
            // Сдвигаем результат влево, освобождая место для нового бита
            result <<= 1;
            // Берем последний бит числа n и добавляем его в результат
            result |= (n & 1);
            // Сдвигаем n вправо (логический сдвиг >>>), чтобы перейти к следующему биту
            n >>>= 1;
        }
        return result;
    }
}
