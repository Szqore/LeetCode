class Solution { //task 28
    public int strStr(String haystack, String needle) {
        // Если needle пустая строка, по условию задачи возращаем 0
        if (needle.isEmpty()) {
            return 0;
    }
    int haystackLen = haystack.length();
    int needleLen = needle.length();
    // Если needle длинные haystack, он точно не можект содержаться
    if (needleLen > haystackLen) {
        return -1;
}

    // Проходим по haystack ,но только до позиции,где needle может поместиться
    for (int i = 0; i <= haystackLen - needleLen; i++) {
        // Проверяем, совпадает ли подстрока с needle
    if (haystack.substring(i, i + needleLen).equals(needle)) {
        return i; //Возращаем индекс первого вхождения
        }
    }
    return -1; // Если не нашли
    }
}
