/**
 * Definition for a binary tree node.
 * public class TreeNode {
 *     int val;
 *     TreeNode left;
 *     TreeNode right;
 *     TreeNode() {}
 *     TreeNode(int val) { this.val = val; }
 *     TreeNode(int val, TreeNode left, TreeNode right) {
 *         this.val = val;
 *         this.left = left;
 *         this.right = right;
 *     }
 * }
 */
class Solution { //task 101
    public boolean isSymmetric(TreeNode root) {
        // Пустое дерево симметрично, но по условию root != null, проверка на всякий случай
        if (root == null) return true;
        // Запускаем рекурсивную проверку левого и правого поддерева
        return isMirror(root.left, root.right);
    }
    
    private boolean isMirror(TreeNode t1, TreeNode t2) {
        // Если оба узла null — зеркальны на этом пути
        if (t1 == null && t2 == null) return true;
        // Если один из узлов null — не зеркальны
        if (t1 == null || t2 == null) return false;
        // Значения должны совпадать, и левое с правым, и правое с левым
        return (t1.val == t2.val) 
            && isMirror(t1.left, t2.right) 
            && isMirror(t1.right, t2.left);
    }
}
