import java.util.HashSet;
import java.util.Set;
//task 349
class Solution {
    public int[] intersection(int[] nums1, int[] nums2) {
        // 1. Добавляем все элементы первого массива в HashSet для быстрого поиска
        Set<Integer> set1 = new HashSet<>();
        for (int num : nums1) {
            set1.add(num);
        }

        // 2. Проверяем элементы второго массива и сохраняем общие в другой Set
        Set<Integer> intersect = new HashSet<>();
        for (int num : nums2) {
            if (set1.contains(num)) {
                intersect.add(num);
            }
        }

        // 3. Конвертируем результат обратно в массив int[]
        int[] result = new int[intersect.size()];
        int i = 0;
        for (int num : intersect) {
            result[i++] = num;
        }

        return result;
    }
}
