class Solution { //task 169
    public int majorityElement(int[] nums) {
        int candidate = 0;
        int count = 0;
        
        // Проходим по массиву, "голосуем" за элемент
        for (int num : nums) {
            if (count == 0) {
                candidate = num;
            }
            
            if (num == candidate) {
                count++;
            } else {
                count--;
            }
        }
        
        // Кандидат гарантированно является большинствующим элементом
        return candidate;
    }
}
