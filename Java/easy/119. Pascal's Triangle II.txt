class Solution { //task 119
    public List<Integer> getRow(int rowIndex) {
        List<Integer> res = new ArrayList<>();
        long current = 1;
        for (int i = 0; i <= rowIndex; i++) {
            res.add((int) current);
            // Вычисляем следующий элемент строки на основе текущего
            current = current * (rowIndex - i) / (i + 1);
        }
        return res;
    }
}
