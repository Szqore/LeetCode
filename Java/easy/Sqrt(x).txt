class Solution { //task 69
    public int mySqrt(int x) {
        if (x < 2 ) return x; // 0 и 1 возвращаем как есть

        int left = 1;
        int right = x / 2; // корень из x не может быть больше x/2 (кроме x=1)

        while (left <= right) {
            int mid = left + (right - left) / 2;

            // Используем long, чтобы избежать переполнения
            long square = (long) mid * mid;

            if (square == x) {
                return mid;
            } else if (square < x) {
                left = mid + 1;
            } else {
                right = mid - 1;
        }   
    }
    return right; // right  это округлённое вниз значение
    }   
}
