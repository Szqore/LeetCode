class Solution { //task 344
    public void reverseString(char[] s) {
        int left = 0;
        int right = s.length - 1;

        while (left < right) {
            // Меняем местами символы
            char temp = s[left];
            s[left] = s [right];
            s[right] = temp;

            // сдивагаем указатели
            left++;
            right--;
        }       
    }
}
