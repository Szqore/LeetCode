import java.util.Stack;
//task 232
class MyQueue {
    private Stack<Integer> input = new Stack<>();
    private Stack<Integer> output = new Stack<>();

    public MyQueue() {
        
    }
    
    // Добавление: O(1)
    public void push(int x) {
        input.push(x);
    }
    
    // Извлечение: амортизированное O(1)
    public int pop() {
        peek(); // Гарантируем, что output содержит элементы
        return output.pop();
    }
    
    // Просмотр первого элемента: амортизированное O(1)
    public int peek() {
        if (output.isEmpty()) {
            while (!input.isEmpty()) {
                output.push(input.pop());
            }
        }
        return output.peek();
    }
    
    // Проверка на пустоту: O(1)
    public boolean empty() {
        return input.isEmpty() && output.isEmpty();
    }
}
