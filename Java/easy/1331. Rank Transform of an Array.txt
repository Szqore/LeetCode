import java.util.Arrays;
import java.util.HashMap;
import java.util.Map;
//task 1331
class Solution {
    public int[] arrayRankTransform(int[] arr) {
        // Создаем и сортируем копию массива, чтобы получить уникальные значения по порядку
        int[] sortedArr = arr.clone();
        Arrays.sort(sortedArr);
        
        // Используем HashMap для хранения соответствия числа и его ранга
        Map<Integer, Integer> rankMap = new HashMap<>();
        int rank = 1;
        
        for (int num : sortedArr) {
            // Если числа еще нет в мапе, добавляем его и увеличиваем ранг
            if (!rankMap.containsKey(num)) {
                rankMap.put(num, rank++);
            }
        }
        
        // Заполняем результирующий массив рангами
        int[] result = new int[arr.length];
        for (int i = 0; i < arr.length; i++) {
            result[i] = rankMap.get(arr[i]);
        }
        
        return result;
    }
}
